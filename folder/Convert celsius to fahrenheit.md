```python
# Convert celsius to fahrenheit scale.
celsius = float(input("Enter temperature in Celsius :"))
fahrenheit = (celsius * 1.8) + 32
print('%0.1f Celsius is equal to %0.1f Fahrenheit' %(celsius,fahrenheit))
```
